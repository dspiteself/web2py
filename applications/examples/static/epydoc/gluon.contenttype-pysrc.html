179 -def contenttype(filename): 

180      """ 
181      returns the Content-Type: string given the extension in the filename. 
182      """ 
183   
184      i = filename.rfind('.') 
185      if i < 0: 
186          return 'text/plain' 
187      return CONTENT_TYPE.get(filename[i:].lower(), 'text/plain')
